Option Explicit

' ----- Lifecycle -----------------------------------------------------------

Private Sub UserForm_Initialize()
    txt_Tutorial.Text = GetTutorialText()
    txt_CacheFolderPath.Text = "C:\BNC_CacheFolder\"

    ' Jezeli setup byl juz zrobiony (np. powtarzamy konfiguracje) - prefill.
    If mod_UserCacheSync.IsSetupCompleted() Then
        txt_Imie.Text = CStr(mod_UserCacheSync.GetUserField("Imie"))
        txt_Nazwisko.Text = CStr(mod_UserCacheSync.GetUserField("Nazwisko"))
        txt_EmailHandlowca.Text = CStr(mod_UserCacheSync.GetUserField("EmailHandlowca"))
        txt_CNA.Text = CStr(mod_UserCacheSync.GetUserField("CNA_HandlowcaID"))
        txt_NrOddzialu.Text = CStr(mod_UserCacheSync.GetUserField("NrOddzialu"))
        txt_EmailKierownika.Text = CStr(mod_UserCacheSync.GetUserField("EmailKierownika"))
        txt_EmailBNC.Text = CStr(mod_UserCacheSync.GetUserField("EmailBNC"))
        Dim folderPath As String
        folderPath = CStr(mod_UserCacheSync.GetUserField("CacheFolderPath"))
        If Len(folderPath) > 0 Then txt_CacheFolderPath.Text = folderPath

        Dim dontShow As Variant
        dontShow = mod_UserCacheSync.GetUserField("DontShowTutorial")
        chk_DontShowTutorial.Value = (CStr(dontShow) = "True")
    End If
End Sub

' ----- Buttons -------------------------------------------------------------

Private Sub btn_Browse_Click()
    Dim folderPath As String
    folderPath = SelectFolder()
    If Len(folderPath) > 0 Then txt_CacheFolderPath.Text = folderPath
End Sub

Private Sub btn_Save_Click()
    ' 1. Zbierz dane z pol
    Dim userData As Object
    Set userData = CreateObject("Scripting.Dictionary")
    userData("Imie") = Trim$(txt_Imie.Text)
    userData("Nazwisko") = Trim$(txt_Nazwisko.Text)
    userData("EmailHandlowca") = Trim$(txt_EmailHandlowca.Text)
    userData("CNA_HandlowcaID") = Trim$(txt_CNA.Text)
    userData("NrOddzialu") = Trim$(txt_NrOddzialu.Text)
    userData("EmailKierownika") = Trim$(txt_EmailKierownika.Text)
    userData("EmailBNC") = Trim$(txt_EmailBNC.Text)
    userData("CacheFolderPath") = Trim$(txt_CacheFolderPath.Text)
    userData("DataRejestracji") = Now()
    userData("SetupCompleted") = True
    userData("DontShowTutorial") = CBool(chk_DontShowTutorial.Value)

    ' 2. Walidacja
    Dim errMsg As String
    errMsg = mod_Validation.ValidateSetupData(userData)
    If Len(errMsg) > 0 Then
        MsgBox errMsg, vbExclamation, "Blad walidacji"
        Exit Sub
    End If

    ' 3. Folder cache - jesli nie istnieje, utworz
    On Error GoTo FolderError
    mod_Utils.EnsureFolderExists CStr(userData("CacheFolderPath"))
    On Error GoTo 0

    ' 4. Zapis do ws_UserCache + auto-sync do xlsx
    mod_UserCacheSync.SaveUserData userData

    ' 5. Auto-recreate plikow cache (DataCache jeszcze nie istnieje przy pierwszym setupie)
    mod_DataCacheSync.EnsureCacheFileExists

    mod_Utils.LogInfo "Setup zakonczony pomyslnie - user=" & CStr(userData("EmailHandlowca"))

    ' 6. Zamknij setup, otworz frm_Main.
    ' TODO(M3): odkomentuj 'frm_Main.Show' i usun MsgBox po implementacji frm_Main.
    Me.Hide
    MsgBox "Setup zakonczony pomyslnie." & vbCrLf & vbCrLf & _
           "Formularz glowny (frm_Main) bedzie dodany w M3.", _
           vbInformation, "BNC_Sender"
    'frm_Main.Show
    Exit Sub

FolderError:
    MsgBox "Nie udalo sie utworzyc folderu cache:" & vbCrLf & _
           CStr(userData("CacheFolderPath")) & vbCrLf & vbCrLf & _
           "Blad: " & Err.Description, vbExclamation, "Blad folderu cache"
End Sub

Private Sub btn_Cancel_Click()
    If MsgBox("Czy na pewno przerwac konfiguracje? Aplikacja nie uruchomi sie poprawnie.", _
              vbYesNo + vbQuestion, "Potwierdzenie") = vbYes Then
        Me.Hide
    End If
End Sub

' ----- Helpers -------------------------------------------------------------

Private Function SelectFolder() As String
    Dim fd As Object
    Set fd = Application.FileDialog(msoFileDialogFolderPicker)
    fd.Title = "Wybierz folder cache"
    fd.AllowMultiSelect = False
    If fd.Show = -1 Then
        SelectFolder = fd.SelectedItems(1)
    Else
        SelectFolder = ""
    End If
End Function
