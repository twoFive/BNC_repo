' ============================================================================
'  ThisWorkbook - kod do wklejenia w istniejacej klasie ThisWorkbook w VBE.
'  UWAGA: ThisWorkbook to specjalny modul klasy generowany przez Excel
'  - nie da sie go zaimportowac jako nowy. Otworz w VBE drzewo
'  Microsoft Excel Objects -> ThisWorkbook (dwuklik), wklej PONIZSZY kod
'  zastepujac ewentualnie istniejacy.
'
'  Routing (M3.3 multi-user):
'  - GetUsersCount = 0 -> pierwszy raz -> frm_Setup (bez pickera)
'  - GetUsersCount >= 1 -> frm_UserPicker (wybor / dodanie nowego)
'
'  Patrz: BNC_Sender_PlanWdrozenia_FazaA.md (M2.3, M3.3)
'         Notatki/DECISIONS.md (ADR-008 - multi-user Registry)
' ============================================================================

Option Explicit

Private Sub Workbook_Open()
    On Error GoTo ErrorHandler

    ' 1. Auto-recreate plikow cache jesli nie istnieja (no-op gdy
    '    CacheFolderPath nie ustawiony - przed pierwszym setupem).
    mod_UserCacheSync.EnsureCacheFileExists
    mod_DataCacheSync.EnsureCacheFileExists

    ' 2. Routing multi-user (M3.3):
    '    - 0 userow -> pierwszy raz -> setup nowego usera
    '    - 1+ userow -> picker (nawet dla 1 usera - Q4 decyzja: swiadome
    '      dzialanie, user musi kliknac zeby "wejsc")
    If mod_UserCacheSync.GetUsersCount() = 0 Then
        mod_Utils.LogInfo "Workbook_Open: pierwszy raz (Registry pusty), pokazuje frm_Setup"
        frm_Setup.Show
    Else
        mod_Utils.LogInfo "Workbook_Open: " & mod_UserCacheSync.GetUsersCount() & _
                         " uzytkownikow w Registry, pokazuje frm_UserPicker"
        frm_UserPicker.Show
    End If
    Exit Sub

ErrorHandler:
    mod_Utils.LogError "ThisWorkbook.Workbook_Open", Err.Number, Err.Description
    MsgBox "Blad podczas startu aplikacji: " & Err.Description, _
           vbExclamation, "BNC_Sender"
End Sub
